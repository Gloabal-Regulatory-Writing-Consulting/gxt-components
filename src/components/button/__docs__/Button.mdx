# Button

Button component with different props.

#### Example

<Canvas of={Button.Primary} />

## Usage

```ts
import { Button } from '@gloabal-regulatory-writing-consulting/gxt-components';;

const ExampleButton = () => {
 const handleClick = (event: React.MouseEvent) => {
    // Handle button click
  };

  return (
    <Button
      onClick={handleClick}
      disabled={false}
      variant="primary"
    >
      Click me
    </Button>
  );
};

export default ExampleButton;
```

#### Arguments

- **disabled** -  A boolean indicating whether the button should be disabled.
- **variant** `"primary" | "secondary" | "negative" | "positive"` - The style variant of the button.
- **onClick** - Function called when the button is clicked. It receives a MouseEventHandler for handling the click event on the button element.
