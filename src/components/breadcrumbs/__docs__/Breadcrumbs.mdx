# BreadcrumbsExample

This is an example component that showcases the usage of `Breadcrumbs` component.

## Usage

```tsx
import Breadcrumbs from "./Breadcrumbs";

const isLinkActive = (path: string) => {
  // set active string logic
}

const breadCrumbList = [
  {
    label: 'Catalog',
    to: '/',
  },
  {
    label: 'View Complaint',
    to: '/1/view-complaint',
  },
];

const BreadcrumbsExample = () => (
  <div style={{ width: "20rem", color: "#115873" }}>
    <Breadcrumbs items={breadCrumbList} isLinkActive={isLinkActive}/>
  </div>
);

export default BreadcrumbsExample;
```
