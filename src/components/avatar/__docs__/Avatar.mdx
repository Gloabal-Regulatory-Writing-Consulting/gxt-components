## Usage

```tsx
import Avatar from "sld-ui";

const Example = () => {
  const onClickHandler = () => {
    // Handle on click of avatar
  };

  const handleImageUpload = (e) => {
    // Handle image upload
  };

  return (
    <Avatar
      size="small"
      user={{ firstName: "John", lastName: "Doe", avatar: "avatar.jpg" }}
      isExpanded={false}
      imageUrl="image.jpg"
      onClickHandler={onClickHandler}
      handleImageUpload={handleImageUpload}
    />
  );
};

export default Example;
```
