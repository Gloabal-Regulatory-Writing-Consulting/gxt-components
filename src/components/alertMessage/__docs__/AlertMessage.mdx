# AlertMessageExample


Example for usage of `AlertMessage` component.

## Usage

```tsx
import AlertMessage from "./AlertMessage";

const alertMessageExampleProps = {
  alertMessage: "Your changes have been saved successfully.",
  alertLink: "/details",
  linkText: "View details",
  alertType: "positive",
};

const AlertMessageExample = () => (
  <div style={{ width: "20rem", color: "#115873" }}>
    <AlertMessage {...alertMessageExampleProps} />
  </div>
);

export default AlertMessageExample;

```
